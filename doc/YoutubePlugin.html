def get(url)
    video = @client.video_by(url)
    url.gsub(%rhttps?:\/\/(www.)?youtube.(fr|com)\/watch\?v=|youtu\.be/, 'https://www.youtube.com')
    {:title => video.title, :author => video.author.name, :url => url}
end
